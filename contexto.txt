CREATE TABLE `tipos_vehiculo` (
  `id_tipo_vehiculo` INT(11) NOT NULL AUTO_INCREMENT,
  `nombre_tipo` VARCHAR(50) NOT NULL UNIQUE,
  PRIMARY KEY (`id_tipo_vehiculo`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;

CREATE TABLE `parqueaderos` (
  `id_parqueadero` INT(11) NOT NULL AUTO_INCREMENT,
  `numero_espacio` VARCHAR(10) NOT NULL UNIQUE,
  `id_tipo_vehiculo_permitido` INT(11) NOT NULL,
  PRIMARY KEY (`id_parqueadero`),
  CONSTRAINT `fk_parqueaderos_tipo_vehiculo` FOREIGN KEY (`id_tipo_vehiculo_permitido`) REFERENCES `tipos_vehiculo` (`id_tipo_vehiculo`) ON UPDATE CASCADE ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;

CREATE TABLE `parqueos` (
  `id_parqueo` INT(11) NOT NULL AUTO_INCREMENT,
  `placa` VARCHAR(10) NOT NULL,
  `id_parqueadero` INT(11) NOT NULL,
  `id_tipo_vehiculo_real` INT(11) NOT NULL,
  `hora_entrada` DATETIME NOT NULL,
  `hora_salida` DATETIME DEFAULT NULL,
  PRIMARY KEY (`id_parqueo`),
  UNIQUE KEY `uk_parqueo_activo` (`id_parqueadero`, `hora_salida`),
  KEY `idx_parqueos_placa` (`placa`),
  KEY `idx_parqueos_hora_entrada` (`hora_entrada`),
  CONSTRAINT `fk_parqueos_parqueadero` FOREIGN KEY (`id_parqueadero`) REFERENCES `parqueaderos` (`id_parqueadero`) ON UPDATE CASCADE ON DELETE RESTRICT,
  CONSTRAINT `fk_parqueos_tipo_vehiculo_real` FOREIGN KEY (`id_tipo_vehiculo_real`) REFERENCES `tipos_vehiculo` (`id_tipo_vehiculo`) ON UPDATE CASCADE ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;

CREATE TABLE `roles` (
  `id_rol` INT(11) NOT NULL AUTO_INCREMENT,
  `nombre_rol` VARCHAR(50) NOT NULL UNIQUE,
  PRIMARY KEY (`id_rol`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;

CREATE TABLE `usuarios` (
  `id_usuario` INT(11) NOT NULL AUTO_INCREMENT,
  `nombre_usuario` VARCHAR(50) NOT NULL UNIQUE,
  `email` VARCHAR(100) NOT NULL UNIQUE,
  `password_hash` VARCHAR(255) NOT NULL,
  `nombre_completo` VARCHAR(100) DEFAULT NULL,
  `fecha_registro` DATETIME DEFAULT CURRENT_TIMESTAMP,
  `activo` BOOLEAN DEFAULT TRUE,
  `id_rol` INT(11) NOT NULL,
  PRIMARY KEY (`id_usuario`),
  KEY `idx_usuarios_email` (`email`),
  CONSTRAINT `fk_usuarios_rol` FOREIGN KEY (`id_rol`) REFERENCES `roles` (`id_rol`) ON UPDATE CASCADE ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;
